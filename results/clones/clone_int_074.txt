

/************ 000 : gcd *************/

int gcd(int m, int n)
{
  if (m == n)
    return m;

  if (m > n)
    return gcd(m - n, n);
  else
    return gcd(m, n - m);

}



/************ 001 : recGcd *************/

int recGcd(int a, int b)
{
  while (a != b)
  {
    if (a > b)
      return recGcd(a - b, b);
    else
      return recGcd(a, b - a);

  }

  return a;
}
